cmake_minimum_required(VERSION 2.6)
project(GUDHIDensityUT)

if(ANN_FOUND)
  include_directories(${ANN_INCLUDE_DIR})

  if (GCOVR_PATH)
    # for gcovr to make coverage reports - Corbera Jenkins plugin
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fprofile-arcs -ftest-coverage")
    set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -fprofile-arcs -ftest-coverage")	
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -fprofile-arcs -ftest-coverage")
  endif()
  if (GPROF_PATH)
    # for gprof to make coverage reports - Jenkins
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -pg")
    set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -pg")
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -pg")
  endif()

  add_executable ( anngraphdensityUT ann_graph_density_unit_test.cpp )
  target_link_libraries(anngraphdensityUT ${ANN_LIBRARIES} ${Boost_SYSTEM_LIBRARY} ${Boost_UNIT_TEST_FRAMEWORK_LIBRARY})

  # Unitary tests
  add_test(NAME anngraphdensityUT
    COMMAND ${CMAKE_CURRENT_BINARY_DIR}/anngraphdensityUT 
      # XML format for Jenkins xUnit plugin 
      --log_format=XML --log_sink=${CMAKE_SOURCE_DIR}/anngraphdensityUT.xml --log_level=test_suite --report_level=no)

else()
  message("ANN_INCLUDE_DIR NOT FOUND")
endif()
