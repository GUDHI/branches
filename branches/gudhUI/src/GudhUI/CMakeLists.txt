	set( QT_USE_QTMAIN   TRUE )

		view/Viewer_instructor.cpp				
		view/Viewer.cpp
		${uis}

	target_link_libraries( GudhUI ${QT_LIBRARIES} ${QGLVIEWER_LIBRARIES} )
